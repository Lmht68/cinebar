28 int NframesFromInterval(const int frame_count,
29 const double interval,
30 const double fps)
31 {
32 if (interval <= 0.0)
33 throw std::invalid_argument("video_processor: Interval must be greater than 0");

35 throw std::runtime_error("video_processor: Invalid FPS value");
36 if (interval < 1.0 / fps)
37 throw std::invalid_argument("video_processor: Interval too small for video FPS");
38 if (frame_count <= 0.0)

41 double duration = frame_count / fps;
42 return static_cast<int>(std::round(duration / interval));

47 std::optional<cinebar_types::VideoBounds> DetectBounds(const cv::Mat &frame_grayed)
48 {
49 // downscale for faster processing
50 cv::Mat frame_downscaled;
51 cv::resize(frame_grayed, frame_downscaled, cv::Size(), kDownScaleFactor, kDownScaleFactor, cv::INTER_AREA);
52
53 int h_downscaled = frame_downscaled.rows;
54 int w_downscaled = frame_downscaled.cols;
55
56 std::vector<double> row_black_ratio(h_downscaled, 0.0);
57 std::vector<double> col_black_ratio(w_downscaled, 0.0);
58
59 // count black pixels
60 for (int y = 0; y < h_downscaled; y++)
61 {
62 for (int x = 0; x < w_downscaled; x++)
63 {
64 if (frame_downscaled.at<uchar>(y, x) < kDefaultThreshold)
65 {
66 row_black_ratio[y] += 1;
67 col_black_ratio[x] += 1;
68 }
69 }
70 }
71
72 for (int y = 0; y < h_downscaled; y++)
73 row_black_ratio[y] /= w_downscaled;
74
75 for (int x = 0; x < w_downscaled; x++)
76 col_black_ratio[x] /= h_downscaled;
77
78 std::vector<bool> black_rows(h_downscaled);
79 std::vector<bool> black_cols(w_downscaled);
80
81 for (int i = 0; i < h_downscaled; i++)
82 black_rows[i] = row_black_ratio[i] > kDefaultMinBlackRatio;
83
84 for (int i = 0; i < w_downscaled; i++)
85 black_cols[i] = col_black_ratio[i] > kDefaultMinBlackRatio;
86
87 int top = 0;
88 while (top < h_downscaled && black_rows[top])
89 top++;
90
91 int bottom = h_downscaled - 1;
92 while (bottom >= 0 && black_rows[bottom])
93 bottom--;
94
95 int left = 0;
96 while (left < w_downscaled && black_cols[left])
97 left++;
98
99 int right = w_downscaled - 1;
100 while (right >= 0 && black_cols[right])
101 right--;
102
103 if (left >= right || top >= bottom)
104 return std::nullopt;
105
106 cinebar_types::VideoBounds bounds{
107 static_cast<int>(left / kDownScaleFactor),
108 static_cast<int>(top / kDownScaleFactor),
109 static_cast<int>(right / kDownScaleFactor),
110 static_cast<int>(bottom / kDownScaleFactor)};
111
112 return bounds;
113 }

126 bool DetermineVideoBounds(cinebar_types::VideoInfo &video_info,
127 cinebar_types::VideoBounds &bounds)
128 {
129 cv::VideoCapture &cap = video_info.capture;
130 double interval = static_cast<double>(video_info.frame_count) / kDefaultSampleFrames;
131 std::vector<cinebar_types::VideoBounds> detections;
132 cv::Mat frame;
133
134 for (int i = 0; i < kDefaultSampleFrames; i++)
135 {
136 int frame_id = static_cast<int>((i + 0.5) * interval);
137 cap.set(cv::CAP_PROP_POS_FRAMES, frame_id);
138 if (!cap.read(frame))
139 continue;
140 cv::Mat gray;
141 cv::cvtColor(frame, gray, cv::COLOR_BGR2GRAY);
142 auto detected = DetectBounds(gray);
143
144 if (detected)
145 detections.push_back(*detected);
146 }
147
148 if (detections.empty())
149 return false;
150
151 std::vector<int> lefts, tops, rights, bottoms;
152
153 for (auto &b : detections)
154 {
155 lefts.push_back(b.left);
156 tops.push_back(b.top);
157 rights.push_back(b.right);
158 bottoms.push_back(b.bottom);
159 }
160
161 auto median = [](std::vector<int> &v)
162 {
163 std::sort(v.begin(), v.end());
164 return v[v.size() / 2];
165 };
166
167 cinebar_types::VideoBounds median_bounds{
168 median(lefts),
169 median(tops),
170 median(rights),
171 median(bottoms)};
172
173 int crop_w = median_bounds.right - median_bounds.left;
174 int crop_h = median_bounds.bottom - median_bounds.top;
175
176 if (crop_w > video_info.width * kMinCropRatio &&
177 crop_h > video_info.height * kMinCropRatio)
178 return false;
179
180 bounds = median_bounds;
181 return true;
182 }

184 void DetectVideoBoxType(cinebar_types::VideoInfo &video_info)
185 {
186 cinebar_types::VideoBounds bounds;
187 bool has_bounds = !DetermineVideoBounds(video_info, bounds);
188 if (!has_bounds)
189 return;
190 video_info.bounds = bounds;
191 int crop_w = bounds.right - bounds.left;
192 int crop_h = bounds.bottom - bounds.top;
193 bool letterbox = crop_h < video_info.height;
194 bool pillarbox = crop_w < video_info.width;
195
196 if (letterbox && pillarbox)
197 video_info.box_type = cinebar_types::BoxType::Windowbox;
198 else if (letterbox)
199 video_info.box_type = cinebar_types::BoxType::Letterbox;
200 else if (pillarbox)
201 video_info.box_type = cinebar_types::BoxType::Pillarbox;
202 else
203 video_info.box_type = cinebar_types::BoxType::None;
204 }
